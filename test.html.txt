    <script>
        const questions = [
            {
                step: "Fase 1: Rastro de Obsessão",
                question: "O que você costuma ensinar para as pessoas sem precisar de um roteiro?",
                options: [
                    { text: "Habilidades técnicas e estéticas (Como criar algo)", type: "Aptidão" },
                    { text: "Princípios morais e espirituais (Por que fazemos)", type: "Intencionalidade" },
                    { text: "Soluções práticas para problemas do dia a dia", type: "Utilidade" }
                ]
            },
            {
                step: "Fase 2: Trindade Vocacional",
                question: "Onde está seu maior incômodo hoje?",
                options: [
                    { text: "Sinto que faço bem, mas não vejo sentido eterno", type: "Tédio" },
                    { text: "Tenho muita vontade, mas me falta técnica", type: "Ansiedade" },
                    { text: "Tenho técnica e sentido, mas não ganho dinheiro com isso", type: "Escassez" }
                ]
            },
            {
                step: "Fase 3: O Custo do Sacrifício",
                question: "Qual 'dor' você aceita carregar com mais resiliência?",
                options: [
                    { text: "A exaustão física da execução (Mãos na massa)", type: "Executor" },
                    { text: "O peso mental da liderança e decisões", type: "Líder" },
                    { text: "O isolamento do estudo e da criação", type: "Criativo" }
                ]
            }
        ];


        let currentStep = 0;
        let scores = { Aptidão: 0, Intencionalidade: 0, Utilidade: 0, Tédio: 0, Ansiedade: 0, Escassez: 0 };


        function startTest() {
            document.getElementById('intro-screen').classList.add('hidden');
            document.getElementById('quiz-screen').classList.remove('hidden');
            showQuestion();
        }


        function showQuestion() {
            const q = questions[currentStep];
            document.getElementById('step-indicator').innerText = q.step;
            document.getElementById('question-text').innerText = q.question;
            document.getElementById('progress-bar').style.width = `${((currentStep) / questions.length) * 100}%`;


            const container = document.getElementById('options-container');
            container.innerHTML = '';


            q.options.forEach(opt => {
                const btn = document.createElement('button');
                btn.className = "w-full text-left p-4 rounded-xl border-2 border-gray-100 hover:border-indigo-500 hover:bg-indigo-50 transition duration-200 group";
                btn.innerHTML = `<span class="text-gray-700 group-hover:text-indigo-700 font-medium">${opt.text}</span>`;
                btn.onclick = () => handleAnswer(opt.type);
                container.appendChild(btn);
            });
        }


        function handleAnswer(type) {
            scores[type]++;
            currentStep++;
            if (currentStep < questions.length) {
                showQuestion();
            } else {
                showResult();
            }
        }


        function showResult() {
            document.getElementById('quiz-screen').classList.add('hidden');
            document.getElementById('result-screen').classList.remove('hidden');
            
            let title = "";
            let desc = "";
            let action = "";


            if (scores.Tédio > 0) {
                title = "O Arquiteto em Busca de Alento";
                desc = "Você possui as ferramentas (skills), mas seu trabalho atual parece desconectado de algo maior. Sua técnica é excelente, mas falta o 'sopro de vida'.";
                action = "Tente integrar um elemento do seu ministério ou fé em um projeto comercial esta semana. Consagre a estética ao Criador.";
            } else if (scores.Ansiedade > 0) {
                title = "O Mensageiro em Preparo";
                desc = "Seu coração já sabe o 'porquê', mas suas mãos ainda estão aprendendo o 'como'. A ansiedade vem do desejo de realizar sem ter a base técnica consolidada.";
                action = "Foque em 1 hora de estudo técnico (Game Design ou PNL) antes de tentar executar o grande projeto. Domine a ferramenta.";
            } else {
                title = "O Catalisador de Valor";
                desc = "Você está em uma fase de convergência. Seu desafio agora é escalar sua utilidade para que o mundo (mercado) reconheça o valor do seu propósito.";
                action = "Valide seu propósito criando um protótipo rápido (MVP) e apresentando para 3 pessoas fora do seu círculo comum.";
            }


            document.getElementById('result-title').innerText = title;
            document.getElementById('result-desc').innerText = desc;
            document.getElementById('result-action').innerText = action;
        }
    </script>
